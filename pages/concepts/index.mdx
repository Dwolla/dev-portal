import { breakDown } from "../../app/components/breakpoints";
import conceptsIcon from "../../assets/images/content-images/content-icons/concepts.svg";
import Pages from "../../app/modules/pages";

export const ConceptIndexPageGrid = () => {
  const bannerProps = {
    icon: conceptsIcon,
    topic: "Concepts",
    description:
      "Understand the fundamental building blocks of integrating with the Dwolla Platform",
  };
  const conceptPages = Pages.under("/concepts").filter(
    (concept) => !!concept.concept
  );

    icon: page.concept.icon,
    href: page.id,
  }));
  return (
    <IndexPageGrid unfilteredCards={conceptCards} bannerProps={bannerProps} />
  );
};
