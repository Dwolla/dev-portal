import guidesIcon from "../../assets/images/content-images/content-icons/guides-icon.svg";
import Pages from "../../app/modules/pages";

export const GuideIndexPageGrid = () => {
  const bannerProps = {
    icon: guidesIcon,
    topic: "Guides",
    description:
      "Step-by-step walkthroughs ensure your application is built to best practices",
  };
  const guideOverviewPages = Pages.under("/guides").filter(
    (guide) => !!guide.guide && guide.guide.step === 0
  );
  const guideCards = guideOverviewPages.map((page) => ({
    topic: page.group.title,
    description: page.group.description,
    icon: page.guide.icon,
    category: page.guide.filter,
    href: page.group.id,
  }));
  return (
    <IndexPageGrid unfilteredCards={guideCards} bannerProps={bannerProps} />
  );
};
